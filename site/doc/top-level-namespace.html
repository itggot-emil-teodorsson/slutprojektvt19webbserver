post('/register_values') do
    reg_and_taken = register_values(params)
    if reg_and_taken[2] == false
        if reg_and_taken[0] == true
            if reg_and_taken[1] == false
                redirect('/reg_complete')
            else
                redirect('/username_taken')
            end
        else
            redirect('/reg_failed')
        end
    else
        redirect('/form_empty')
    end
end
